Solution of TDSE for a free gaussian wavepacket using FTCS Explicit Method¶

Implementing tridiagonal matrix method to evaluate solution (with dead wall boundary condition) This can also be solved using traditional array construction as of 1D diffusion equation

>psi[x][t+1]=psi[x][t]+0.5*(i*dt/dx**2)*(psi[x+1][t]-2psi[x][t]+psi[x-1][t])-i*V[x][t]*dt*psi[x][t]

>psi[x][t+1]=mu*psi[x+1][t]+(1-2*mu-i*dt*V[x][t])*psi[x][t]+mu*psi[x-1][t]

>where mu=0.5*i*dt/dx**2

This simple minded algorithm is not a good algorithm as norm is not preserved for approximation and the solution diverges for very small dt (norm increases)

import numpy as np
import matplotlib.pyplot as plt

#Constants
a=1.0         #standard deviation of gaussian
x0=-5.0       #mean of gaussian
k=20.0        #wave-vector

#Parameters
L=20.0        #length of the box
N=1000        #Discretized spatial points
dt=0.00005    #Time step (Neumann Stability Condition)
t=0           #Initial time
tf=0.1        #Final time

#Initial condition
xs,dx=np.linspace(-L/2,L/2,N,retstep=True)
us=np.exp(-(xs-x0)**2/a+(0.0+1.j)*k*xs)    #This is same as the 1st column (initial time) of 2D array used in 1D diffusion equation. This 1D array will be updated with time
mu=dt/dx**2*(0.+1.j)/2

#Applied potential
Vs=0*xs[1:-1]    #Only taking inside points which will be updated by M matrix.                               

#Explicit Algorithm to create tridiagonal matrix M
M=np.eye(N-2)*(1-2*mu)-np.diag(1.j*dt*Vs)+mu*np.eye(N-2,k=1)+mu*np.eye(N-2,k=-1)      #np.eye() creates identity matrix. k parameter shifts the leading diagonal to submajor or subminor position. np.diag() creates a diagonal matrix with elements of Vs array

while t<tf:
    us[1:-1]=np.dot(M,us[1:-1])            #Solution matrix generated over space in a given time slice (except end points) which is being updated with next time slices
    t+=dt

#Plotting
plt.plot(xs,np.abs(us)**2,color='blue')
plt.title(r'Probability Distribution of Free Gaussian wavepacket (explicit method)'+'\n$Time = %10.6f$'%(t-dt))
plt.axvline(x=0,color='black')
plt.axhline(y=0,color='black')
plt.axhline(y=1,color="red")
plt.ylim(-0.5,1.1)
plt.show()

import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import simpson

#Constants
sigma=3         #standard deviation of gaussian
x0=-10.0        #mean of gaussian
k=10.0          #wave-vector

#Parameters
L=100           #length of the box
N=1000          #Discretized spatial points
dt=0.01         #Time step (Neumann Stability Condition)
t=0             #Initial time
tf=5            #Final time

#Initial condition
xs,dx=np.linspace(-L/2,L/2,N,retstep=True)
mu=dt/dx**2*(0.+1.j)/2

a=(np.sqrt(2*np.pi*sigma**2))
us=1/(np.sqrt(a))*np.exp(-(xs-x0)**2/(4*sigma**2)+1j*k*xs)    #This is same as the 1st column (initial time) of 2D array used in 1D diffusion equation. This 1D array will be updated with time

Norm_initial=simpson(np.abs(us)**2,x=xs)
print('The initial norm of the wavefunction is :',Norm_initial)

#Applied potential
Vs=0*xs[1:-1]    #Only taking inside points which will be updated by M matrix.                              

#Seting Matrix
A1=np.eye(N-2)*(1-mu)+np.diag(-1.j*dt/2*Vs)+mu/2*np.eye(N-2,k=1)+mu/2*np.eye(N-2,k=-1)
A2=np.eye(N-2)*(1+mu)+np.diag(1.j*dt/2*Vs)-mu/2*np.eye(N-2,k=1)-mu/2*np.eye(N-2,k=-1)
B=np.dot(np.linalg.inv(A2),A1)

#Time Evolution and Plotting
plt.figure(figsize=(10,5))

plt.plot(xs,us,color='lightskyblue',label=r'Initial Wavepacket $\psi(x,0)$')
plt.plot(xs,np.abs(us)**2,color='peachpuff',label=r'$|\psi(x,0)|^2$')

while t<tf:
    us[1:-1]=np.dot(B,us[1:-1])            #Solution matrix generated over space in a given time slice (except end points) which is being updated with next time slices
    t+=dt

plt.title('Time Evolution of Gaussian wavepacket for free particle\n'+r'Time =%10.6f,norm =%10.6f'%(t,simpson(np.abs(us)**2,x=xs)))

plt.plot(xs,us,color='blue',label=r'Evolving Wavepacket $\psi(x,t)$')   
plt.plot(xs,np.abs(us)**2,color='red',label=r'$|\psi(x,t)|^2$')     

plt.axvline(x=0,color='black',lw=1)
plt.axhline(y=0,color='black',lw=1)
plt.plot(xs[1:-1],Vs,'--',color='green',lw=1,label='Potential')

plt.ylim(-0.5,0.8)
plt.legend()
plt.show()

#Conservation of norm
Norm_final=simpson(np.abs(us)**2,x=xs)
print('The final norm of the wavefunction is :',Norm_final)

if round(Norm_initial/Norm_final)==1:
    print('The Norm is preserved')
else:
    print('Norm is not preserved')

t=0             #Initial time
tf=15           #Final time

#Initial condition

mu=dt/dx**2*(0.+1.j)/2

prefactor=(omega/np.pi)**0.25

print('The initial norm of the wavefunction is :',Norm_initial)

#Applied potential
Vs=np.where((xs[1:-1]>=5) & (xs[1:-1]<=6),35,0)     #Only taking inside points which will be updated by M matrix. In NumPy, multiplying a 2D identity matrix by a 1D array broadcasts that array along the diagonal of the identity matrix.                              

#Seting Matrix
A1=np.eye(N-2)*(1-mu)+np.diag(-1j*dt/2*Vs)+mu/2*np.eye(N-2,k=1)+mu/2*np.eye(N-2,k=-1)
A2=np.eye(N-2)*(1+mu)+np.diag(1j*dt/2*Vs)-mu/2*np.eye(N-2,k=1)-mu/2*np.eye(N-2,k=-1)
B=np.dot(np.linalg.inv(A2),A1)

#Dynamic Plotting
plt.figure(figsize=(10,5))

plt.plot(xs,us,color='skyblue',label=r'Initial Wavepacket $\psi(x,0)$')
plt.plot(xs,np.abs(us)**2,color='khaki',label=r'$|\psi(x,0)|^2$')

while t<tf:
    us[1:-1]=np.dot(B,us[1:-1])            #Solution matrix generated over space in a given time slice (except end points) which is being updated with next time slices
    t+=dt

plt.title('Time Evolution of Gaussian wavepacket undergoing Tunneling\n'+r'Time =%10.6f,norm =%10.6f'%(t,simpson(np.abs(us)**2,x=xs)))

plt.plot(xs,us,color='steelblue',label=r'Evolving Wavepacket $\psi(x,t)$')       
plt.plot(xs,np.abs(us)**2,color='gold',label=r'$|\psi(x,t)|^2$')             

plt.axvline(x=0,color='black',lw=1)
plt.axhline(y=0,color='black',lw=1)
plt.plot(xs[1:-1],Vs,'--',color='red',lw=1,label='Potential')

plt.ylim(-0.5,0.8)
plt.legend()
plt.show()

#Conservation of norm
Norm_final=simpson(np.abs(us)**2,x=xs)
print('The final norm of the wavefunction is :',Norm_final)

if round(Norm_initial/Norm_final)==1:
    print('The Norm is preserved')
else:
    print('Norm is not preserved')
